make_report <- function(rdsname, cutoff_date) {
  tables <- tables_for_report(rdsname, cutoff_date)
  if (length(tables) == 0) {
    a <- knit_expand(text = c("# {{rdsname}}",
                         "Nothing to say.\n"))
  } else {
    a <- knit_expand(text = c("# {{rdsname}}",
                         "## Changes", "{{kable(tables[[1]])}}",
                         "## Games", "{{kable(tables[[2]])}}",
                         "## Odds now", "{{kable(tables[[3]])}}"))
  }
  cat(a, sep = "\n")
}
